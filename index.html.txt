  <script>
    const ws = new WebSocket('ws://localhost:3000');
    const alertMessage = document.getElementById('alertMessage');
    const tamperStatus = document.getElementById('tamperStatus');
    const activityLog = document.getElementById('activityLog');
    const lastUpdate = document.getElementById('lastUpdate');

    ws.onopen = () => {
      console.log('Connected to server');
    };

    ws.onmessage = (event) => {
      const msg = event.data;
      const time = new Date().toLocaleTimeString();
      lastUpdate.textContent = time;

      // Update alert box
      alertMessage.innerHTML = `<div class="alert">${msg}</div>`;
      tamperStatus.textContent = msg.includes('Tamper') ? 'Alert!' : 'Normal';
      tamperStatus.className = msg.includes('Tamper') ? 'status-alert' : 'status-normal';

      // Add to activity log
      const li = document.createElement('li');
      li.textContent = `${time} - ${msg}`;
      activityLog.prepend(li);
    };
  </script>
